032    package com.angel.sdk;
033    
034    import java.util.Calendar;
035    import java.util.GregorianCalendar;
036    
037    
038    /**
039     * Serializes dates into the BLE standard type
040     * org.bluetooth.characteristic.day_date_time
041     */
042    public class BleDayDateTime {
043    
044        public static final int DAY_DATE_SERIALIZED_SIZE = 8;
045        public static final int DATE_SERIALIZED_SIZE = 7;
046    
047    
048        public static GregorianCalendar Deserialize(byte[] data) {
049            if (data.length != DATE_SERIALIZED_SIZE && data.length != DAY_DATE_SERIALIZED_SIZE) {
050                throw new RuntimeException("Trying to deserialize a date from a buffer of wrong size");
051            }
052    
053            int year = data[1] & 0xFF;
054            year = year << 8;
055            year = year | (data[0] & 0xFF);
056            int month = data[2];
057            int day = data[3];
058            int hour = data[4];
059            int minutes = data[5];
060            int seconds = data[6];
061            return new GregorianCalendar(year, month - 1, day, hour, minutes, seconds);
062        }
063    
064    
065        public static byte[] SerializeDayDateTime(GregorianCalendar dateTime) {
066            int year = dateTime.get(Calendar.YEAR);
067            int month = dateTime.get(Calendar.MONTH);
068            int day = dateTime.get(Calendar.DAY_OF_MONTH);
069            int hour = dateTime.get(Calendar.HOUR_OF_DAY);
070            int minutes = dateTime.get(Calendar.MINUTE);
071            int seconds = dateTime.get(Calendar.SECOND);
072            int dayOfWeek = dateTime.get(Calendar.DAY_OF_WEEK);
073    
074            byte[] data = new byte[DAY_DATE_SERIALIZED_SIZE];
075            data[0] = (byte) (year & 0x00FF);
076            data[1] = (byte) ((year & 0xFF00) >> 8);
077            data[2] = (byte) (month + 1); // GregorianCalendar months are zero based
078            data[3] = (byte) day;
079            data[4] = (byte) hour;
080            data[5] = (byte) minutes;
081            data[6] = (byte) seconds;
082            data[7] = (byte) dayOfWeek;
083    
084            return data;
085        }
086    
087    
088        public static byte[] SerializeDateTime(GregorianCalendar dateTime) {
089            int year = dateTime.get(Calendar.YEAR);
090            int month = dateTime.get(Calendar.MONTH);
091            int day = dateTime.get(Calendar.DAY_OF_MONTH);
092            int hour = dateTime.get(Calendar.HOUR_OF_DAY);
093            int minutes = dateTime.get(Calendar.MINUTE);
094            int seconds = dateTime.get(Calendar.SECOND);
095    
096            byte[] data = new byte[DATE_SERIALIZED_SIZE];
097            data[0] = (byte) (year & 0x00FF);
098            data[1] = (byte) ((year & 0xFF00) >> 8);
099            data[2] = (byte) (month + 1); // GregorianCalendar months are zero based
100            data[3] = (byte) day;
101            data[4] = (byte) hour;
102            data[5] = (byte) minutes;
103            data[6] = (byte) seconds;
104    
105            return data;
106        }
107    
108    }
