032    
033    /**
034     * Thrown when trying to connect to an inaccessible Bluetooth, either because it
035     * doesn't exist on the device or because the application doesn't have the
036     * required permission.
037     */
038    public class BluetoothInaccessibleException extends Exception {
039    
040        public BluetoothInaccessibleException() {
041        }
042    
043    
044        public BluetoothInaccessibleException(String detailMessage) {
045            super(detailMessage);
046        }
047    
048        private static final long serialVersionUID = 1L;
049    }
