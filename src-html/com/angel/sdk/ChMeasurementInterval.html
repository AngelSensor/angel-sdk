032    package com.angel.sdk;
033    
034    import android.bluetooth.BluetoothGattCharacteristic;
035    
036    import java.util.UUID;
037    
038    
039    /** org.bluetooth.characteristic.measurement_interval */
040    public class ChMeasurementInterval extends BleCharacteristic<Integer> {
041        public final static UUID CHARACTERISTIC_UUID = UUID.fromString("00002a21-0000-1000-8000-00805f9b34fb");
042    
043    
044        public ChMeasurementInterval(BluetoothGattCharacteristic gattCharacteristic, BleDevice bleDevice) {
045            super(CHARACTERISTIC_UUID, gattCharacteristic, bleDevice);
046        }
047    
048    
049        public ChMeasurementInterval() {
050            super(CHARACTERISTIC_UUID);
051        }
052    
053    
054        @Override
055        protected Integer processCharacteristicValue() {
056            BluetoothGattCharacteristic c = getBaseGattCharacteristic();
057            return c.getIntValue(BluetoothGattCharacteristic.FORMAT_UINT16, 0);
058        }
059    
060    }
