032    package com.angel.sdk;
033    
034    import android.bluetooth.BluetoothGattCharacteristic;
035    
036    import com.angel.sdk.ChBatteryLevel.BatteryLevelValue;
037    
038    import java.util.UUID;
039    
040    
041    /** org.bluetooth.characteristic.battery_level */
042    public class ChBatteryLevel extends BleCharacteristic<BatteryLevelValue> {
043        public final static UUID CHARACTERISTIC_UUID = UUID.fromString("00002a19-0000-1000-8000-00805f9b34fb");
044    
045        public ChBatteryLevel(BluetoothGattCharacteristic gattCharacteristic,
046                              BleDevice bleDevice) {
047            super(CHARACTERISTIC_UUID, gattCharacteristic, bleDevice);
048        }
049    
050        public ChBatteryLevel() {
051            super(CHARACTERISTIC_UUID);
052        }
053    
054        @Override
055        protected BatteryLevelValue processCharacteristicValue() {
056            BatteryLevelValue batteryLevelValue = new BatteryLevelValue();
057            BluetoothGattCharacteristic c = getBaseGattCharacteristic();
058            batteryLevelValue.value = c.getIntValue(BluetoothGattCharacteristic.FORMAT_UINT8, 0);
059            return batteryLevelValue;
060        }
061    
062        /**
063         * Battery Level is too simple for this abstraction but {@code BleCharacteristic}
064         * requires a class to encapsulate the characteristic fields.
065         */
066        public class BatteryLevelValue {
067            public int value;
068        }
069    }
