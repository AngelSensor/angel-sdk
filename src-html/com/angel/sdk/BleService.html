031    package com.angel.sdk;
032    
033    import android.bluetooth.BluetoothGattCharacteristic;
034    import android.bluetooth.BluetoothGattService;
035    import android.util.Log;
036    
037    import java.lang.reflect.Constructor;
038    import java.lang.reflect.InvocationTargetException;
039    import java.util.HashMap;
040    import java.util.UUID;
041    
042    
043    /**
044     * Base class for all the concrete GATT service classes.
045     */
046    public abstract class BleService {
047    
048        public BleService(UUID uuid, BluetoothGattService gattService, BleDevice bleDevice) {
049            if (!uuid.equals(gattService.getUuid())) {
050                throw new AssertionError();
051            }
052    
053            mUuid = uuid;
054            mBaseGattService = gattService;
055            mBleDevice = bleDevice;
056        }
057    
058    
059        /**
060         * Creates a non-operation instance used to investigate static properties in
061         * contexts where these properties are accessible only via a class instance.
062         * Calling most methods of such an object will cause an undefined behavior.
063         */
064        public BleService(UUID uuid) {
065            mUuid = uuid;
066            mBaseGattService = null;
067            mBleDevice = null;
068        }
069    
070    
071        public UUID getUuid() {
072            return mUuid;
073        }
074    
075    
076        public BleCharacteristic<?> getCharacteristic(UUID uuid) {
077            return mBleCharacteristics.get(uuid);
078        }
079    
080    
081        public BluetoothGattService getBaseGattService() {
082            return mBaseGattService;
083        }
084    
085    
086        /**
087         * Register a concrete characteristic class that encapsulates one of the
088         * characteristics supported by the service.
089         * <p>
090         * Immediately after the registration creates an instance and returns it.
091         * The concrete characteristic class has to follow certain rules. If one of
092         * the rules is not met, an exception is thrown.
093         * 
094         * @param characteristicClass
095         *            the concrete characteristic class
096         * @throws InstantiationException
097         *             failed to create an instance of the concrete characteristic
098         * @throws IllegalAccessException
099         *             the constructor of the concrete characteristic class is not
100         *             accessible
101         * @throws NoSuchMethodException
102         *             a concrete characteristic must define two constructors: one
103         *             without arguments and another with two arguments of types
104         *             {@code BluetoothGattCharacteristic} and {@code BleDevice}
105         * @throws InvocationTargetException
106         *             one of the constructors on the concrete characteristic threw
107         *             an exception
108         */
109        protected <T extends BleCharacteristic<?>> T createAndRegisterCharacteristic(Class<T> characteristicClass)
110                throws InstantiationException,
111                    IllegalAccessException,
112                    NoSuchMethodException,
113                    InvocationTargetException {
114    
115            // Create a non-operational dummy instance only to access the UUID
116            T dummyCh = characteristicClass.newInstance();
117            UUID uuid = dummyCh.getUuid();
118    
119            // Check whether the service indeed supports the characteristic
120            BluetoothGattCharacteristic baseCh = mBaseGattService.getCharacteristic(uuid);
121            if (baseCh == null) {
122                Log.e("ang", "No characteristic " + uuid + " in service " + mBaseGattService.getUuid());
123                return null;
124            }
125    
126            // Was the characteristic registered earlier?
127            if (mBleCharacteristics.containsKey(uuid)) {
128                throw new RuntimeException("Trying to register characteristic (UUID:"
129                                           + uuid.toString() + ") twice inside a service (UUID:"
130                                           + mBaseGattService.getUuid() + ")");
131            }
132    
133            Constructor<T> ctor = characteristicClass.getConstructor(BluetoothGattCharacteristic.class,
134                                                                     BleDevice.class);
135            T characteristic = ctor.newInstance(baseCh, mBleDevice);
136    
137            mBleCharacteristics.put(uuid, characteristic);
138    
139            return characteristic;
140        }
141    
142        private final UUID mUuid;
143        private final HashMap<UUID, BleCharacteristic<?>> mBleCharacteristics = new HashMap<UUID, BleCharacteristic<?>>();
144        private final BluetoothGattService mBaseGattService;
145        private final BleDevice mBleDevice;
146    }
