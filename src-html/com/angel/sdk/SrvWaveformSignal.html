032    package com.angel.sdk;
033    
034    import android.bluetooth.BluetoothGattService;
035    
036    import java.lang.reflect.InvocationTargetException;
037    import java.util.UUID;
038    
039    public class SrvWaveformSignal extends BleService {
040        public final static UUID SERVICE_UUID = UUID.fromString("481d178c-10dd-11e4-b514-b2227cce2b54");
041    
042    
043        public SrvWaveformSignal(BluetoothGattService gattService, BleDevice bleDevice) {
044            super(SERVICE_UUID, gattService, bleDevice);
045    
046            try {
047    
048                // Register all the concrete classes for all the characteristics 
049                // of Heart Rate Service. Failing one of the assertions bellow means
050                // we have mistakes with the definition of one of the characteristics
051                mOpticalWaveform = createAndRegisterCharacteristic(ChOpticalWaveform.class);
052                mAccelerationWaveform = createAndRegisterCharacteristic(ChAccelerationWaveform.class);
053            } catch (InstantiationException e) {
054                throw new AssertionError();
055            } catch (IllegalAccessException e) {
056                throw new AssertionError();
057            } catch (NoSuchMethodException e) {
058                throw new AssertionError();
059            } catch (IllegalArgumentException e) {
060                throw new AssertionError();
061            } catch (InvocationTargetException e) {
062                throw new AssertionError();
063            }
064            
065            if (mOpticalWaveform == null || mAccelerationWaveform == null) {
066                throw new AssertionError();
067            }
068        }
069    
070    
071        public SrvWaveformSignal() {
072            super(SERVICE_UUID);
073        }
074    
075    
076        /** Get access to Optical Waveform characteristic */
077        public ChOpticalWaveform getOpticalWaveform() {
078            return mOpticalWaveform;
079        }
080    
081        
082        /** Get access to Acceleration Waveform characteristic */
083        public ChAccelerationWaveform getAccelerationWaveform() {
084            return mAccelerationWaveform;
085        }
086        
087        
088        private ChOpticalWaveform mOpticalWaveform;
089        private ChAccelerationWaveform mAccelerationWaveform;
090    
091    }
