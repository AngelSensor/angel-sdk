031    package com.angel.sdk;
032    
033    import android.bluetooth.BluetoothGattCharacteristic;
034    
035    import java.util.UUID;
036    
037    
038    /**
039     * Base class for all the concrete characteristic classes.
040     */
041    public abstract class BleCharacteristic<ValueType> {
042    
043        public BleCharacteristic(UUID uuid,
044                                 BluetoothGattCharacteristic gattCharacteristic,
045                                 BleDevice bleDevice) {
046            if (!uuid.equals(gattCharacteristic.getUuid())) {
047                throw new AssertionError();
048            }
049            mUuid = uuid;
050            mBaseGattCharacteristic = gattCharacteristic;
051            mBleDevice = bleDevice;
052        }
053    
054    
055        /**
056         * Creates a non-operation instance used to investigate static properties in
057         * contexts where these properties are accessible only via a class instance.
058         * Calling most methods of such an object will cause an undefined behavior.
059         */
060        public BleCharacteristic(UUID uuid) {
061            mUuid = uuid;
062            mBaseGattCharacteristic = null;
063            mBleDevice = null;
064        }
065    
066    
067        public UUID getUuid() {
068            return mUuid;
069        }
070    
071    
072        public BluetoothGattCharacteristic getBaseGattCharacteristic() {
073            return mBaseGattCharacteristic;
074        }
075    
076    
077        public BleDevice getBleDevice() {
078            return mBleDevice;
079        }
080    
081    
082        /** Used to report characteristic value asynchronously */
083        public interface ValueReadyCallback<T> {
084            public void onValueReady(T value);
085        }
086    
087    
088        /**
089         * Enable notifications for the characteristic. The actual notifications are
090         * asynchronous and are delivered via the supplied callback.
091         */
092        public void enableNotifications(ValueReadyCallback<ValueType> callback) {
093            mOnDataReadyListener = callback;
094            mBleDevice.enableCharacteristicNotifications(this.mBaseGattCharacteristic);
095        }
096    
097        /**
098         * Request characteristic value. The value is returned asynchronously via
099         * the supplied callback.
100         */
101        public void readValue(ValueReadyCallback<ValueType> callback) {
102            mOnDataReadyListener = callback;
103            mBleDevice.readCharacteristic(this.mBaseGattCharacteristic);
104        }
105    
106    
107        protected final void onCharacteristicChanged() {
108            ValueType value = processCharacteristicValue();
109            mOnDataReadyListener.onValueReady(value);
110        };
111    
112    
113        /**
114         * Converts the raw characteristic value into the human friendly ValueType.
115         * <p>
116         * For example, a heart rate measurement may contain both the heart rate
117         * value and several RR-intervals. BluetoothGattCharacteristic returns the
118         * value as a byte array. The purpose of this method is to parse the byte
119         * array and create a strongly typed data structure.
120         * <p>
121         * The implementation is expected to call
122         * {@link #getBaseGattCharacteristic()} to access the underlying
123         * characteristic and use one of its {@code get*Value} methods to get the
124         * raw characteristic value.
125         */
126        protected abstract ValueType processCharacteristicValue();
127    
128        private final UUID mUuid;
129        private final BluetoothGattCharacteristic mBaseGattCharacteristic;
130        private final BleDevice mBleDevice;
131        private ValueReadyCallback<ValueType> mOnDataReadyListener;
132    
133    }
