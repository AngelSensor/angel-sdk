031    package com.angel.sdk;
032    
033    import android.bluetooth.BluetoothGattService;
034    
035    import java.lang.reflect.InvocationTargetException;
036    import java.util.UUID;
037    
038    
039    /** GATT Battery Service */
040    public class SrvBattery extends BleService {
041        public final static UUID SERVICE_UUID = UUID.fromString("0000180f-0000-1000-8000-00805f9b34fb");
042    
043    
044        public SrvBattery(BluetoothGattService gattService, BleDevice bleDevice) {
045            super(SERVICE_UUID, gattService, bleDevice);
046    
047            try {
048    
049                // Register all the concrete classes for all the characteristics
050                // of Battery Level Service. Failing one of the assertions bellow
051                // would indicate incorrect definition of one of the characteristics
052                mChBatteryLevel = createAndRegisterCharacteristic(ChBatteryLevel.class);
053    
054            } catch (InstantiationException e) {
055                throw new AssertionError();
056            } catch (IllegalAccessException e) {
057                throw new AssertionError();
058            } catch (NoSuchMethodException e) {
059                throw new AssertionError();
060            } catch (IllegalArgumentException e) {
061                throw new AssertionError();
062            } catch (InvocationTargetException e) {
063                throw new AssertionError();
064            }
065        }
066    
067        public SrvBattery() {
068            super(SERVICE_UUID);
069        }
070    
071        /** Get access to Battery Level characteristic */
072        public ChBatteryLevel getBatteryLevel() {
073            return mChBatteryLevel;
074        }
075    
076        private ChBatteryLevel mChBatteryLevel;
077    }
