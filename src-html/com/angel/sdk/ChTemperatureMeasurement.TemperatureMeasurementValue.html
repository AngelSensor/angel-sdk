032    package com.angel.sdk;
033    
034    import android.bluetooth.BluetoothGattCharacteristic;
035    
036    import com.angel.sdk.ChTemperatureMeasurement.TemperatureMeasurementValue;
037    
038    import java.util.Arrays;
039    import java.util.GregorianCalendar;
040    import java.util.UUID;
041    
042    
043    public class ChTemperatureMeasurement extends BleCharacteristic<TemperatureMeasurementValue> {
044        public final static UUID CHARACTERISTIC_UUID = UUID.fromString("00002a1c-0000-1000-8000-00805f9b34fb");
045    
046        public final static int CELSIUS = 0;
047        public final static int FAHRENHEIT = 1;
048    
049    
050        public ChTemperatureMeasurement(BluetoothGattCharacteristic gattharacteristic,
051                                        BleDevice bleDevice) {
052            super(CHARACTERISTIC_UUID, gattharacteristic, bleDevice);
053        }
054    
055    
056        public ChTemperatureMeasurement() {
057            super(CHARACTERISTIC_UUID);
058        }
059    
060    
061        protected ChTemperatureMeasurement(UUID uuid,
062                                           BluetoothGattCharacteristic gattharacteristic,
063                                           BleDevice bleDevice) {
064            super(uuid, gattharacteristic, bleDevice);
065        }
066    
067    
068        protected ChTemperatureMeasurement(UUID uuid) {
069            super(uuid);
070        }
071    
072    
073        @Override
074        protected TemperatureMeasurementValue processCharacteristicValue() {
075            TemperatureMeasurementValue tv = new TemperatureMeasurementValue();
076            BluetoothGattCharacteristic c = getBaseGattCharacteristic();
077            int nextOffset = 0;
078            int flags = c.getIntValue(BluetoothGattCharacteristic.FORMAT_UINT8, nextOffset);
079            nextOffset += 1;
080    
081            tv.mTemperatureUnits = flags & 0x1;
082            tv.mTemperatureMeasurement = c.getFloatValue(BluetoothGattCharacteristic.FORMAT_FLOAT,
083                                                         nextOffset);
084            nextOffset += 4;
085    
086            if ((flags & 0x2) != 0) {
087                byte[] dateArray = Arrays.copyOfRange(c.getValue(), nextOffset,
088                                                      nextOffset + BleDayDateTime.DATE_SERIALIZED_SIZE);
089                tv.mTimeStamp = BleDayDateTime.Deserialize(dateArray);
090                nextOffset += BleDayDateTime.DATE_SERIALIZED_SIZE;
091            }
092    
093            if ((flags & 0x4) != 0) {
094                tv.mTemperatureType = c.getIntValue(BluetoothGattCharacteristic.FORMAT_UINT8,
095                                                    nextOffset);
096            }
097            return tv;
098        }
099    
100    
101        public class TemperatureMeasurementValue {
102            int mTemperatureUnits;
103            Float mTemperatureMeasurement;
104            GregorianCalendar mTimeStamp;
105            int mTemperatureType;
106    
107    
108            public int getTemperatureUnits() {
109                return mTemperatureUnits;
110            }
111    
112    
113            public Float getTemperatureMeasurement() {
114                return mTemperatureMeasurement;
115            }
116    
117    
118            public GregorianCalendar getTimeStamp() {
119                return mTimeStamp;
120            }
121    
122    
123            public int getTemperatureType() {
124                return mTemperatureType;
125            }
126        }
127    }
