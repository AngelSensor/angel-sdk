031    package com.angel.sdk;
032    
033    import android.app.Activity;
034    import android.bluetooth.BluetoothAdapter;
035    import android.bluetooth.BluetoothDevice;
036    import android.content.pm.PackageManager;
037    import android.os.Handler;
038    
039    import java.util.TreeSet;
040    
041    
042    /**
043     * Scans for available Bluetooth devices.
044     * <p>
045     * This class gives a concise interface for Bluetooth devices discovery. Each
046     * discovered device is reported via a callback object configured in the
047     * constructor.
048     */
049    public class BleScanner {
050    
051        /**
052         * {@code BleScanner} reports about the discovered devices using this
053         * interface.
054         * <p>
055         * Each device is reported only once.
056         */
057        public interface ScanCallback {
058            public void onBluetoothDeviceFound(BluetoothDevice device);
059        }
060    
061    
062        /**
063         * Creates and configures a new {@code BleScanner} instance.
064         * 
065         * @param activity
066         *            the Android activity that owns the object
067         * @param scanCallback
068         *            the callback object that will handle the device discovery
069         * @throws BluetoothInaccessibleException
070         *             thrown if the Bluetooth device is unaccessible
071         */
072        public BleScanner(Activity activity, ScanCallback scanCallback)
073                throws BluetoothInaccessibleException {
074    
075            if (!activity.getPackageManager().hasSystemFeature(PackageManager.FEATURE_BLUETOOTH_LE)) {
076                throw new BluetoothInaccessibleException();
077            }
078    
079            mBluetoothAdapter = BluetoothAdapter.getDefaultAdapter();
080            if (mBluetoothAdapter == null) {
081                throw new BluetoothInaccessibleException();
082            }
083    
084            mUserScanCallback = scanCallback;
085            mActivity = activity;
086            mScanHandler = new Handler();
087        }
088    
089    
090        /**
091         * Scan for 10 seconds, report discovered devices, stop scanning.
092         */
093        public void startScan() {
094            mScanStopper = new Runnable() {
095                @Override
096                public void run() {
097                    stopScan();
098                }
099            };
100            mScanHandler.postDelayed(mScanStopper, SCAN_PERIOD_MILLIS);
101            mBluetoothAdapter.startLeScan(mBleAdapterScanCallback);
102        }
103    
104    
105        public void stopScan() {
106            mScanHandler.removeCallbacks(mScanStopper);
107            mBluetoothAdapter.stopLeScan(mBleAdapterScanCallback);
108        }
109    
110        /**
111         * This callback is called each time BluetoothAdapter#startLeScan()
112         * discovers a device. Note that it might be called more than once for a
113         * device.
114         */
115        private final BluetoothAdapter.LeScanCallback mBleAdapterScanCallback = new BluetoothAdapter.LeScanCallback() {
116            @Override
117            public void onLeScan(final BluetoothDevice device, int rssi, byte[] scanRecord) {
118            if (!mDeviceAddresses.contains(device.getAddress())) {
119                mDeviceAddresses.add(device.getAddress());
120    
121                mActivity.runOnUiThread(new Runnable() {
122                    @Override
123                    public void run() {
124                    mUserScanCallback.onBluetoothDeviceFound(device);
125                    }
126                });
127            }
128            }
129        };
130    
131        private static final long SCAN_PERIOD_MILLIS = 10000;
132        private final BluetoothAdapter mBluetoothAdapter;
133    
134        private final Activity mActivity;
135        private final ScanCallback mUserScanCallback;
136        private final Handler mScanHandler;
137        private Runnable mScanStopper;
138    
139        /** A collection of addresses of all the discovered devices */
140        private final TreeSet<String> mDeviceAddresses = new TreeSet<String>();
141    }
