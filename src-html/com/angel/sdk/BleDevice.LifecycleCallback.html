058    /**
059     * Bluetooth device object. Manages the life cycle and provides access to the
060     * GATT services.
061     * <p>
062     * The client is expected to initialize a BleDevice object in three steps:
063     * create an instance of BleDevice or an inherited class, register all the
064     * required services by calling {@link #registerServiceClass(Class)}, call
065     * {@link #connect(String)}.
066     */
067    public class BleDevice {
068        private static HandlerThread makeThread() {
069            HandlerThread handlerThread = new HandlerThread("BleDevice_callback_thread");
070            handlerThread.start();
071            return handlerThread;
072        }
073    
074        public BleDevice(Context context, LifecycleCallback lifecycleCallback) {
075            this(context, lifecycleCallback, new Handler(makeThread().getLooper()));
076        }
077    
078        public BleDevice(Context context, LifecycleCallback lifecycleCallback, Handler callbackHandler) {
079            mActivity = (Activity) context;
080            mLifecycleCallback = lifecycleCallback;
081            mCallbackHandler = callbackHandler;
082        }
083    
084        /**
085         * Interface for notifications on BLE device life cycle events
086         */
087        public interface LifecycleCallback {
088    
089            /** Called after the device discovers new supported services */
090            public void onBluetoothServicesDiscovered(BleDevice device);
091    
092    
093            /** Called on device disconnection */
094            public void onBluetoothDeviceDisconnected();
095    
096            public void onReadRemoteRssi(final int rssi);
097        }
098    
099    
100        public void connect(String deviceAddress) {
101    
102            mDeviceAddress = deviceAddress;
103            if (mBleController == null) {
104                mAction = Action.CONNECT;
105                Intent gattServiceIntent = new Intent(mActivity.getApplicationContext(),
106                                                      BleController.class);
107                boolean gotBinded = mActivity.getApplicationContext().bindService(gattServiceIntent,
108                                                                                  mServiceConnection,
109                                                                                  Context.BIND_AUTO_CREATE);
110                if (!gotBinded) throw new RuntimeException("Failed to bind to BleController");
111            } else {
112                connect(mBleController.getRemoteDevice(deviceAddress));
113            }
114        }
115        
116        
117        public void disconnect() {
118            mDeviceAddress = "";
119            if (mBluetoothGatt != null) mBluetoothGatt.close();
120        }
121    
122        public BleController getBleController() {
123            return mBleController;
124        }
125        
126    
127        /**
128         * Get a concrete service object of the specified type as was previously
129         * registered using {@link BleDevice#registerServiceClass(UUID, Class)}.
130         * Throws ClassCastException if for a specific service UUID was registered
131         * one class but then getService was called with another.
132         * 
133         * Usage example:
134         * <code> SrvHeartRate hr = device.getService(SrvHeartRate.class); </code>
135         * 
136         * @return null if the service wasn't found
137         */
138        public <T extends BleService> T getService(Class<T> serviceClass) {
139            try {
140                BleService service = serviceClass.newInstance();
141    
142                @SuppressWarnings("unchecked")
143                T concreteService = (T) mBleServices.get(service.getUuid());
144                return concreteService;
145            } catch (InstantiationException e) {
146                throw new AssertionError(); // Should have been verified in registerServiceClass
147            } catch (IllegalAccessException e) {
148                throw new AssertionError(); // Should have been verified in registerServiceClass
149            }
150        }
151    
152    
153        /**
154         * Registers a service. If a service class is not registered, it won't be
155         * accessible via {@link #getService(Class)}. Note that a service class can
156         * be registered even if the actual device eventually doesn't support the
157         * service. This method exists to allows definition of new services outside
158         * of the SDK.
159         */
160        public void registerServiceClass(Class<? extends BleService> serviceClass)
161                throws NoSuchMethodException,
162                    IllegalAccessException,
163                    InstantiationException {
164            // Create a non-operational dummy instance only to access the UUID
165            BleService service = serviceClass.newInstance();
166    
167            // Check that the class has the required public constructor
168            Constructor<? extends BleService> ctor = serviceClass.getConstructor(BluetoothGattService.class,
169                                                                                 BleDevice.class);
170            if (!Modifier.isPublic(ctor.getModifiers())) {
171                throw new IllegalAccessException(serviceClass.getName() + " constructor must be public");
172            }
173            mBleServiceClasses.put(service.getUuid(), serviceClass);
174        }
175    
176    
177        /**
178         * Request characteristic value. The value is returned asynchronously via
179         * ACTION_DATA_READ_AVAILABLE intent.
180         */
181        public void readCharacteristic(final BluetoothGattCharacteristic characteristic) {
182            if (characteristic == null) {
183                return;
184            }
185            final int properties = characteristic.getProperties();
186            if ((mBluetoothGatt != null)
187                && (properties & BluetoothGattCharacteristic.PROPERTY_READ) > 0) {
188                mBluetoothGatt.readCharacteristic(characteristic);
189            }
190        }
191    
192    
193        public void writeCharacteristic(BluetoothGattCharacteristic characteristic) {
194            if (characteristic == null) {
195                return;
196            }
197            final int properties = characteristic.getProperties();
198            if ((mBluetoothGatt != null)
199                && (properties & BluetoothGattCharacteristic.PROPERTY_WRITE) > 0) {
200                mBluetoothGatt.writeCharacteristic(characteristic);
201            }
202        }
203    
204    
205        /**
206         * Enable notifications and/or indications for the characteristic. Both
207         * types are enabled in the rare case when a characteristic supports them.
208         * 
209         * This method will block if called on the BLE thread. Make sure it is called from
210         * another thread.
211         */
212        public void enableCharacteristicNotifications(final BluetoothGattCharacteristic characteristic) {
213            if (characteristic == null) {
214                return;
215            }
216            final int properties = characteristic.getProperties();
217            if (mBluetoothGatt != null) {
218                if (((properties & BluetoothGattCharacteristic.PROPERTY_NOTIFY) > 0) ||
219                    ((properties & BluetoothGattCharacteristic.PROPERTY_INDICATE) > 0)) {
220                    setCharacteristicNotification(characteristic, true);
221                }
222            }
223        }
224    
225        public void readRemoteRssi() {
226            if (mBluetoothGatt != null) {
227                mBluetoothGatt.readRemoteRssi();
228            }
229        }
230    
231        private void connect(BluetoothDevice device) {
232            mBluetoothGatt = device.connectGatt(mBleController, false, mGattCallback);
233        }
234        
235        
236        private void setCharacteristicNotification(BluetoothGattCharacteristic characteristic,
237                                                   boolean enabled) {
238            if (mBluetoothGatt == null) {
239                return;
240            }
241    
242            if (!mBluetoothGatt.setCharacteristicNotification(characteristic, enabled)) {
243                throw new AssertionError("Failed setCharacteristicNotification for UUID " + characteristic.getUuid());
244            }
245    
246            BluetoothGattDescriptor descriptor = characteristic.getDescriptor(UUID.fromString(BleController.CLIENT_CHARACTERISTIC_CONFIG));
247            final byte NOTIFY_AND_INDICATE[] = {3,0};
248            descriptor.setValue(enabled ? NOTIFY_AND_INDICATE : BluetoothGattDescriptor.DISABLE_NOTIFICATION_VALUE);
249            mWaitingForConfirmation = true;
250            if (!mBluetoothGatt.writeDescriptor(descriptor)) {
251                throw new AssertionError("Failed to write BLE descriptor " + descriptor.getUuid() + " for UUID " + characteristic.getUuid());
252            }
253    
254            try {
255                synchronized (this) {
256                    wait(DESCRIPTOR_WRITE_TIMEOUT);
257                    if (mWaitingForConfirmation) {
258                        throw new AssertionError("Did not receive confirmation for mBluetoothGatt.writeDescriptor(" + characteristic.getUuid() + ")");
259                    }
260                }
261            } catch (InterruptedException e) {
262                throw new AssertionError("Interrupted while waiting for response to mBluetoothGatt.writeDescriptor");        
263            }
264        }
265        
266        
267        private void addNewServices(List<BluetoothGattService> supportedGattServices) {
268            for (BluetoothGattService bluetoothGattService : supportedGattServices) {
269                BleService bleService = null;
270                UUID serviceUuid = bluetoothGattService.getUuid();
271                Class<? extends BleService> serviceClass = mBleServiceClasses.get(serviceUuid);
272    
273                // Is the service registered?
274                if (serviceClass == null) continue;
275    
276                // Was the service discovered earlier?
277                if (mBleServices.containsKey(serviceUuid)) continue;
278    
279                try {
280                    Constructor<? extends BleService> ctor = serviceClass.getConstructor(BluetoothGattService.class,
281                                                                                         BleDevice.class);
282                    bleService = ctor.newInstance(bluetoothGattService, this);
283                } catch (InstantiationException e) {
284                    throw new AssertionError(); // Should have been verified in registerServiceClass
285                } catch (IllegalAccessException e) {
286                    throw new AssertionError(); // Should have been verified in registerServiceClass
287                } catch (NoSuchMethodException e) {
288                    throw new AssertionError(); // Should have been verified in registerServiceClass
289                } catch (InvocationTargetException e) {
290                    Log.e(TAG, "Could not create an instance of " + serviceClass.getName()
291                               + ". Constructor threw an exception");
292                }
293    
294                mBleServices.put(bluetoothGattService.getUuid(), bleService);
295            }
296        }

299        private enum Action {
300            IDLE, CONNECT, DISCONNECT
301        }
302    
303        private final ServiceConnection mServiceConnection = new ServiceConnection() {
304            @Override
305            public void onServiceConnected(ComponentName componentName, IBinder service) {
306                mBleController = ((BleController.LocalBinder) service).getService();
307                if (!mBleController.initialize()) {
308                    return;
309                }
310                switch (mAction) {
311                case CONNECT:
312                    connect(mBleController.getRemoteDevice(mDeviceAddress));
313                    break;
314                //TODO possibly deprecated
315                case DISCONNECT:
316                    mBluetoothGatt.close();
317                    break;
318                default:
319                    break;
320                }
321                mAction = Action.IDLE;
322            }
323    
324    
325            @Override
326            public void onServiceDisconnected(ComponentName componentName) {
327                mBluetoothGatt = null;
328                mBleController = null;
329            }
330        };
331    
332    
333        private final BluetoothGattCallback mGattCallback = new BluetoothGattCallback() {
334            @Override
335            public void onConnectionStateChange(BluetoothGatt gatt, int status, int newState) {
336                if (newState == BluetoothProfile.STATE_CONNECTED) {
337                    mBluetoothGatt.discoverServices();
338    
339                } else if (newState == BluetoothProfile.STATE_DISCONNECTED) {
340                    if (mLifecycleCallback != null) {
341                        mCallbackHandler.post(new Runnable() {
342                            @Override
343                            public void run() {
344                                mLifecycleCallback.onBluetoothDeviceDisconnected();
345                            }
346                        });
347                    }
348                }
349            }
350    
351            @Override
352            public void onServicesDiscovered(BluetoothGatt gatt, int status) {
353                if (status == BluetoothGatt.GATT_SUCCESS) {
354                    addNewServices(mBluetoothGatt.getServices());
355                    if (mLifecycleCallback != null) {
356                        mCallbackHandler.post(new Runnable() {
357                            @Override
358                            public void run() {
359                                mLifecycleCallback.onBluetoothServicesDiscovered(BleDevice.this);
360                            }
361                        });
362                    }
363                } else {
364                    //TODO
365                }
366            }
367    
368            @Override
369            public void onCharacteristicRead(BluetoothGatt gatt,
370                                             BluetoothGattCharacteristic characteristic,
371                                             int status) {
372                if (status == BluetoothGatt.GATT_SUCCESS) {
373                    String serviceUuid = characteristic.getService().getUuid().toString();
374                    if (serviceUuid == null) throw new AssertionError();
375    
376                    String charcteristicUuid = characteristic.getUuid().toString();
377                    if (charcteristicUuid == null) throw new AssertionError();
378                    handleOnCharacteristicChanged(serviceUuid, charcteristicUuid);
379                }
380            }
381    
382            @Override
383            public void onCharacteristicChanged(BluetoothGatt gatt,
384                                                BluetoothGattCharacteristic characteristic) {
385                String serviceUuid = characteristic.getService().getUuid().toString();
386                if (serviceUuid == null) throw new AssertionError();
387    
388                String charcteristicUuid = characteristic.getUuid().toString();
389                if (charcteristicUuid == null) throw new AssertionError();
390                handleOnCharacteristicChanged(serviceUuid, charcteristicUuid);
391            }
392    
393            @Override
394            public void onDescriptorWrite(BluetoothGatt gatt, BluetoothGattDescriptor descriptor, int status) {
395                synchronized (BleDevice.this) {
396                    mWaitingForConfirmation = false;
397                    BleDevice.this.notify();
398                }
399            }
400    
401            @Override
402            public void onReadRemoteRssi(BluetoothGatt gatt, final int rssi, int status) {
403                if (status == BluetoothGatt.GATT_SUCCESS && mLifecycleCallback != null) {
404                    mCallbackHandler.post(new Runnable() {
405                        @Override
406                        public void run() {
407                            mLifecycleCallback.onReadRemoteRssi(rssi);
408                        }
409                    });
410                }
411            }
412        };
413        
414    
415        private void handleOnCharacteristicChanged(String serviceUuid, String charcteristicUuid) {
416            final BleService bleService = mBleServices.get(UUID.fromString(serviceUuid));
417            final BleCharacteristic<?> bleCharacteristic = bleService.getCharacteristic(UUID.fromString(charcteristicUuid));
418            mCallbackHandler.post(new Runnable() {
419                @Override
420                public void run() {
421                    bleCharacteristic.onCharacteristicChanged();
422                }
423            });
424        }
425    
426        /** Maximum milliseconds to wait for writeDescriptor() confirmation from the remote device */
427        private final int DESCRIPTOR_WRITE_TIMEOUT = 5000;
428    
429        private BluetoothGatt mBluetoothGatt;
430        private Action mAction = Action.IDLE;
431        private String mDeviceAddress;
432        private final Activity mActivity;
433        private final LifecycleCallback mLifecycleCallback;
434        private BleController mBleController = null;
435        private final HashMap<UUID, BleService> mBleServices = new HashMap<UUID, BleService>();
436        private final HashMap<UUID, Class<? extends BleService>> mBleServiceClasses = new HashMap<UUID, Class<? extends BleService>>();
437        private int mRssi;
438    
439        /** Thread for asynchronous execution of callbacks */
440        private HandlerThread mCallbackThread;
441    
442        /**
443         * Handler for asynchronous execution of callbacks. If user code makes new
444         * calls to Bluetooth services from within a callback, execution will be
445         * blocked. Deferring callbacks to a separate thread resolves this.
446         */
447        private Handler mCallbackHandler;
448    
449        private boolean mWaitingForConfirmation = false;
450        
451        private static final String TAG = BleDevice.class.getName();
