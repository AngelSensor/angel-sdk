032    package com.angel.sdk;
033    
034    import android.bluetooth.BluetoothGattService;
035    
036    import java.lang.reflect.InvocationTargetException;
037    import java.util.UUID;
038    
039    
040    /** GATT Health Thermometer Service */
041    public class SrvHealthThermometer extends BleService {
042        public final static UUID SERVICE_UUID = UUID.fromString("00001809-0000-1000-8000-00805f9b34fb");
043    
044    
045        public SrvHealthThermometer(BluetoothGattService gattService, BleDevice bleDevice) {
046            super(SERVICE_UUID, gattService, bleDevice);
047            try {
048                mTemperatureMeasurement = createAndRegisterCharacteristic(ChTemperatureMeasurement.class);
049                mChTemperatureType = createAndRegisterCharacteristic(ChTemperatureType.class);
050                mChMeasurementInterval = createAndRegisterCharacteristic(ChMeasurementInterval.class);
051                mIntermediateTemperature = createAndRegisterCharacteristic(ChIntermediateTemperature.class);
052            } catch (InstantiationException e) {
053                throw new AssertionError();
054            } catch (IllegalAccessException e) {
055                throw new AssertionError();
056            } catch (NoSuchMethodException e) {
057                throw new AssertionError();
058            } catch (InvocationTargetException e) {
059                throw new AssertionError();
060            }
061    
062        }
063    
064    
065        public SrvHealthThermometer() {
066            super(SERVICE_UUID);
067        }
068    
069    
070        /** Get access to Temperature Measurement characteristic */
071        public ChTemperatureMeasurement getTemperatureMeasurement() {
072            return mTemperatureMeasurement;
073        }
074    
075    
076        /** Get access to Temperature Type characteristic */
077        public ChTemperatureType getChTemperatureType() {
078            return mChTemperatureType;
079        }
080    
081    
082        /** Get access to Measurement Interval characteristic */
083        public ChMeasurementInterval getmChMeasurementInterval() {
084            return mChMeasurementInterval;
085        }
086    
087    
088        /** Get access to Intermediate Temperature characteristic */
089        public ChIntermediateTemperature getIntermediateTemperature() {
090            return mIntermediateTemperature;
091        }
092    
093        private ChTemperatureMeasurement mTemperatureMeasurement;
094        private ChTemperatureType mChTemperatureType;
095        private ChMeasurementInterval mChMeasurementInterval;
096        private ChIntermediateTemperature mIntermediateTemperature;
097    }
