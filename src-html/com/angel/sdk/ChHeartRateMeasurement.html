032    package com.angel.sdk;
033    
034    import android.bluetooth.BluetoothGattCharacteristic;
035    
036    import com.angel.sdk.ChHeartRateMeasurement.HeartRateMeasurementValue;
037    
038    import java.util.UUID;
039    
040    
041    /** org.bluetooth.characteristic.heart_rate_measurement */
042    public class ChHeartRateMeasurement extends BleCharacteristic<HeartRateMeasurementValue> {
043        public final static UUID CHARACTERISTIC_UUID = UUID.fromString("00002a37-0000-1000-8000-00805f9b34fb");
044    
045        public ChHeartRateMeasurement(BluetoothGattCharacteristic gattCharacteristic,
046                                      BleDevice bleDevice) {
047            super(CHARACTERISTIC_UUID, gattCharacteristic, bleDevice);
048        }
049    
050        public ChHeartRateMeasurement() {
051            super(CHARACTERISTIC_UUID);
052        }
053    
054        @Override
055        protected HeartRateMeasurementValue processCharacteristicValue() {
056            HeartRateMeasurementValue heartRateMeasurementValue = new HeartRateMeasurementValue();
057            BluetoothGattCharacteristic c = getBaseGattCharacteristic();
058            int flags = c.getIntValue(BluetoothGattCharacteristic.FORMAT_UINT8, 0);
059            int nextOffset = 1;
060            int format = -1;
061            if ((flags & 0x01) != 0) {
062                format = BluetoothGattCharacteristic.FORMAT_UINT16;
063                nextOffset += 2;
064            } else {
065                format = BluetoothGattCharacteristic.FORMAT_UINT8;
066                nextOffset += 1;
067            }
068            heartRateMeasurementValue.mHeartRateMeasurement = c.getIntValue(format, 1);
069            if ((flags & 0x08) != 0) {
070    
071                heartRateMeasurementValue.mEnergyExpended = c.getIntValue(BluetoothGattCharacteristic.FORMAT_UINT16,
072                                                                          nextOffset);
073                nextOffset += 2;
074            }
075    
076            if ((flags & 0x10) != 0) {
077    
078                int rrCount = (c.getValue().length - nextOffset) / 2;
079                heartRateMeasurementValue.mRRIntervals = new int[rrCount];
080                for (int i = 0; i < rrCount; i++) {
081                    heartRateMeasurementValue.mRRIntervals[i] = c.getIntValue(BluetoothGattCharacteristic.FORMAT_UINT16,
082                                                                              nextOffset);
083                    nextOffset += 2;
084                }
085            }
086            return heartRateMeasurementValue;
087        }
088    
089    
090        public class HeartRateMeasurementValue {
091            int mHeartRateMeasurement;
092            int mEnergyExpended;
093            int[] mRRIntervals;
094    
095    
096            public int getHeartRateMeasurement() {
097                return mHeartRateMeasurement;
098            }
099    
100    
101            public int getEnergyExpended() {
102                return mEnergyExpended;
103            }
104    
105    
106            public int[] getRRIntervals() {
107                return mRRIntervals;
108            }
109        }
110    }
