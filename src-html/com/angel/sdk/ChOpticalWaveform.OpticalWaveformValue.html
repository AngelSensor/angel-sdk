032    package com.angel.sdk;
033    
034    import android.bluetooth.BluetoothGattCharacteristic;
035    
036    import com.angel.sdk.ChOpticalWaveform.OpticalWaveformValue;
037    
038    import java.util.ArrayList;
039    import java.util.UUID;
040    
041    
042    public class ChOpticalWaveform extends BleCharacteristic<OpticalWaveformValue> {
043        public final static UUID CHARACTERISTIC_UUID = UUID.fromString("334c0be8-76f9-458b-bb2e-7df2b486b4d7");
044    
045    
046        public ChOpticalWaveform(BluetoothGattCharacteristic gattharacteristic,
047                                 BleDevice bleDevice) {
048            super(CHARACTERISTIC_UUID, gattharacteristic, bleDevice);
049        }
050    
051    
052        public ChOpticalWaveform() {
053            super(CHARACTERISTIC_UUID);
054        }
055        
056        
057        @Override
058        protected OpticalWaveformValue processCharacteristicValue() {
059            OpticalWaveformValue result = new OpticalWaveformValue();
060            
061            BluetoothGattCharacteristic ch = getBaseGattCharacteristic();
062            byte[] buffer = ch.getValue();
063            
064            final int TWO_SAMPLES_SIZE = 6;
065            for (int i=TWO_SAMPLES_SIZE-1; i<buffer.length; i+=TWO_SAMPLES_SIZE) {
066                
067                int green = unsignedByte(buffer[i-5]) + 
068                            unsignedByte(buffer[i-4])*256 + 
069                            unsignedByte(buffer[i-3])*256*256;
070                
071                int blue  = unsignedByte(buffer[i-2]) + 
072                            unsignedByte(buffer[i-1])*256 + 
073                            unsignedByte(buffer[i])*256*256;
074            
075                OpticalSample sample = new OpticalSample();
076                sample.green = TwosComplement(green);
077                sample.blue = TwosComplement(blue);
078                result.wave.add(sample);
079            }
080            
081            return result;
082        }
083        
084        
085        public class OpticalSample {
086            public int green;
087            public int blue;
088        }
089        
090        public class OpticalWaveformValue {
091            public ArrayList<OpticalSample> wave = new ArrayList<OpticalSample>();
092        }
093    
094        private static int unsignedByte(byte x) {
095            return x & 0xFF;
096        }
097    
098        int TwosComplement(int raw) {
099            final int BITS = 24;
100            final int NEGATIVE_BITMASK = 1 << (BITS - 1);
101            final int FULL_RANGE = 1 << BITS;
102    
103            int value = raw;
104            if ((value & NEGATIVE_BITMASK) != 0) {
105                value -= FULL_RANGE;
106            }
107            
108            return value;
109        }
110    }
